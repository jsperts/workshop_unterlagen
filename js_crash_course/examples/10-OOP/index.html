function Pet(name, color) {
    this.name = name;
    this.color = color;
  }

  Pet.prototype.getName = function() {
    return this.name;
  };
  Pet.prototype.getColor = function() {
    return this.color;
  };
  var pet = new Pet('I am a pet', 'red');


  function Dog(name, color) {
    Pet.call(this, name, color);
  }

  Dog.prototype = Object.create(Pet.prototype);

  Dog.prototype.getName = function() {
    var name = Pet.prototype.getName.call(this);
    return 'I am: ' + name;
  };

  var skip = new Dog('skip', 'black');

  pet.getName(); // 'I am a pet'
  skip.getName(); // 'I am: skip'
  skip.getColor(); // 'black'
